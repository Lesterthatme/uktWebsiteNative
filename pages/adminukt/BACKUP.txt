  <!-- JavaScript to Load Content Dynamically -->
  <script>
    $(document).ready(function () {
        function loadContent(page) {
            $.ajax({
                url: page, // Load PHP file dynamically
                type: "GET",
                success: function (data) {
                    $(".content_org").html(data); // Replace content with loaded page
                    console.log(page + " loaded!");

                    // Update the URL without reloading the page
                    history.pushState(null, null, "?page=" + page);

                    // Reload all JavaScript files to ensure full functionality
                    reloadScripts();
                },
                error: function () {
                    console.error("Failed to load " + page);
                }
            });
        }

        // Function to reload all JavaScript files after loading new content
        function reloadScripts() {
            let scripts = [
                "../../assets/bootstrap/js/jquery-3.7.1.js",
                "../../assets/bootstrap/js/bootstrap.bundle.min.js",
                "../../assets/bootstrap/js/script.js",
                "../../assets/bootstrap/js/carousel.js",
                "../../assets/bootstrap/js/drag_and_drop.js?v=2.1",
                "../../assets/bootstrap/js/activeLink.js?v=1.7",
                "../../assets/bootstrap/js/logs.js?v=1.4",
                "../../assets/bootstrap/js/site_settings.js?v=1.0"
            ];

            // Remove old scripts to prevent duplication
            $("script.dynamic-script").remove();

            // Append each script dynamically
            scripts.forEach(function (src) {
                let script = document.createElement("script");
                script.src = src;
                script.classList.add("dynamic-script");
                document.body.appendChild(script);
            });
        }

        // Load correct page based on URL on first load
        let urlParams = new URLSearchParams(window.location.search);
        let initialPage = urlParams.get("page") || "about_org.php";
        loadContent(initialPage);

        // Handle tab click
        $(".doc-link").click(function (e) {
            e.preventDefault(); // Prevent default link behavior
            let page = $(this).data("page");
            loadContent(page);

            // Update active state
            $(".doc-link").removeClass("active");
            $(this).addClass("active");
        });
    });
</script>


  </script>
